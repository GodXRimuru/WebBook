rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    
    // Helper functions
    function isAuthenticated() {
      return request.auth != null;
    }
    
    function isOwner(userId) {
      return isAuthenticated() && request.auth.uid == userId;
    }
    
    // Users collection
    match /users/{userId} {
      allow read: if true; // Public profiles
      allow create: if isOwner(userId);
      allow update: if isOwner(userId);
      allow delete: if isOwner(userId);
    }
    
    // Views collection - public read, system writes
    match /views/{viewId} {
      allow read: if true;
      allow write: if true; // Allow anyone to increment views
    }
    
    // Likes/Dislikes collection
    match /likes/{likeId} {
      allow read: if true; // Anyone can see like counts
      allow write: if isAuthenticated(); // Only authenticated users can like/dislike
    }
    
    // Comments collection
    match /comments/{commentId} {
      allow read: if true; // Anyone can read comments
      allow create: if isAuthenticated(); // Only authenticated can comment
      allow update: if isOwner(resource.data.userId); // Only owner can edit
      allow delete: if isOwner(resource.data.userId); // Only owner can delete
    }
    
    // Bookmarks collection
    match /bookmarks/{bookmarkId} {
      allow read: if isOwner(resource.data.userId);
      allow create: if isAuthenticated();
      allow delete: if isOwner(resource.data.userId);
    }
    
    // Progress/Reading History
    match /progress/{progressId} {
      allow read: if isOwner(resource.data.userId);
      allow write: if isAuthenticated();
    }
    
    match /readingHistory/{historyId} {
      allow read: if isOwner(resource.data.userId);
      allow write: if isAuthenticated();
    }
    
    // Highlights collection
    match /highlights/{highlightId} {
      allow read: if isOwner(resource.data.userId);
      allow create: if isAuthenticated();
      allow update: if isOwner(resource.data.userId);
      allow delete: if isOwner(resource.data.userId);
    }
    
    // Polls collection
    match /polls/{pollId} {
      allow read: if true; // Anyone can see polls
      allow write: if isAuthenticated(); // Only authenticated can vote
    }
    
    // Notifications collection
    match /notifications/{notificationId} {
      allow read: if isOwner(resource.data.userId);
      allow write: if true; // System can create notifications
      allow delete: if isOwner(resource.data.userId);
    }
    
    // Chapter scheduling (admin only)
    match /scheduled_chapters/{chapterId} {
      allow read: if true;
      allow write: if isAuthenticated(); // Add admin check here
    }
    
    // Drafts (creator only)
    match /drafts/{draftId} {
      allow read: if isOwner(resource.data.creatorId);
      allow write: if isAuthenticated();
    }
  }
}
